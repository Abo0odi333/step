function doGet(e) { var params = e.parameter; var action = params.action; var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var timestamp = new Date(); var output = {}; // قراءة البيانات الموحدة من السطر الثاني دائماً (أول سطر بيانات بعد العناوين) if (action === "load") { var dataRow = sheet.getRange(2, 1, 1, 6).getValues()[0]; output = { status: "success", checked: JSON.parse(dataRow[0] || "[]"), notes: JSON.parse(dataRow[1] || "{}"), examDate: dataRow[2] || null, isDark: dataRow[3] === true || dataRow[3] === "true", globalWrongQuestions: JSON.parse(dataRow[4] || "{}") }; } // حفظ البيانات الموحدة else if (action === "sync") { sheet.getRange(2, 1).setValue(params.checked || "[]"); sheet.getRange(2, 2).setValue(params.notes || "{}"); sheet.getRange(2, 3).setValue(params.examDate || ""); sheet.getRange(2, 4).setValue(params.isDark || "false"); sheet.getRange(2, 5).setValue(params.globalWrongQuestions || "{}"); sheet.getRange(2, 6).setValue(timestamp); output = { status: "success" }; } // دعم نظام JSONP لتخطي حظر CORS من الأجهزة المختلفة var callback = params.callback; if (callback) { return ContentService.createTextOutput(callback + "(" + JSON.stringify(output) + ")") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(JSON.stringify(output)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { return doGet(e); }
